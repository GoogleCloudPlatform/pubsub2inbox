```hcl
project_id        = "<your-project-id>"
region            = "europe-west1"
slack_token       = "xoxb-...." # Slack OAuth token for workspace
slack_signing_key = "8e................02f" # Slack signing key
slack_app_id      = "A........L" # App ID
```
